175## @fn wait_and_analyze_results()
176## @brief waits for results from asynchronous runs and analyze them when they are ready
177function wait_and_analyze_results() {
178 local n_results_exp=$1
179 local n_results=0
180 local results=""
181 local n_successful=0
182 local is_ok=""
183
184 if [ ! -d $RES_FOLDER ]
185 then
186 return 0
187 fi
188
189 while [ $n_results -lt $n_results_exp ]
190 do
191 echo "Waiting for results: $n_results are ready of $n_results_exp"
192 sleep 2 #//or maybe 2-3?
193 if [ -z "$(ls -A $RES_FOLDER/)" ] #empty
194 then
195 continue
196 fi
197 n_results=`ls $RES_FOLDER/$RES_PREFIX* | wc -w`
198 done
199
200 results=`ls $RES_FOLDER/$RES_PREFIX*`
201 for res in $results
202 do
203 is_ok=`cat $res | grep "OK"`
204 if [ "$is_ok" = "OK" ]
205 then
206 echo "$res $is_ok"
207 n_successful=`expr $n_successful + 1`
208 else
209 echo "$res FAILED"
210 fi
211 done
212
213 if [ $n_successful -eq $n_results_exp ]
214 then
215 echo "OK"
216 return 0
217 fi
218 echo "FAILED: $(expr $n_successful - $n_results_exp)"
219}
