14function wait_and_analyze_results(){
15 local n_results_exp=$1
16 local n_results=0
17 local results=""
18 local n_successful=0
19 local is_ok=""
20
21 if [ ! -d $RES_FOLDER ]
22 then
23 return 0
24 fi
25
26 while [ $n_results -lt $n_results_exp ]
27 do
28 echo "Waiting for results: $n_results are ready of $n_results_exp"
29 sleep 2 #//or maybe 2-3?
30 if [ -z "$(ls -A $RES_FOLDER/)" ] #empty
31 then
32 continue
33 fi
34 n_results=`ls $RES_FOLDER/$RES_PREFIX* | wc -w`
35 done
36
37 results=`ls $RES_FOLDER/$RES_PREFIX*`
38 for res in $results
39 do
40 is_ok=`cat $res | grep "OK"`
41 if [ "$is_ok" = "OK" ]
42 then
43 echo "$res $is_ok"
44 n_successful=`expr $n_successful + 1`
45 else
46 echo "$res FAILED"
47 fi
48 done
49
50 if [ $n_successful -eq $n_results_exp ]
51 then
52 echo "OK"
53 return 0
54 fi
55 echo "FAILED: $(expr $n_successful - $n_results_exp)"
56}

59# Returns NH to destination IP prefix on the given compute.
60# Only the first path (NH) is analyzed
61# Input parameters:
62# - dst_ip: the destination IP prefix
63# - vrf_name: name of VRF instance
64# - comp_ip: the IP of the compute server
65#
66function vr_state(){
67 local dst_ip=$1
68 local vrf_name=$2
69 local comp_ip=$3
70 #request for the list of vrfs:
71 curl -s -o t_vrfs.xml http://$comp_ip:8085/Snh_VrfListReq
72 #get a vrf id
73 local vrf_id=`xmllint --xpath "string(//VrfSandeshData[name='$vrf_name']/ucindex)" t_vrfs.xml`
74 # return if $vrf_id is empty
75 if [ "$vrf_id" = "" ]
76 then
77 echo "Ev"
78 return
79 fi
80
81 curl -s -o t_nhs.xml http://$comp_ip:8085/Snh_Inet4UcRouteReq?x=$vrf_id
82 #echo t_nhs
83 local n_nhs=`xmllint --xpath "count(//RouteUcSandeshData[src_ip='$dst_ip']/path_list/list/PathSandeshData)" t_nhs.xml`
84 if [ "$n_nhs" = "" ]
85 then
86 echo "En"
87 return
88 fi
89 if [ $n_nhs -gt 5 ]
90 then
91 echo "Ec$n_nhs"
92 return
93 fi
94 if [ $n_nhs -eq 0 ]
95 then
96 echo "$comp_ip,$i_nh"
97 return
98 fi
99 i_nh=1
100 while [ $i_nh -le $n_nhs ]
101 do
102 local nh_type=`xmllint --xpath "string(//RouteUcSandeshData[src_ip='$dst_ip']/path_list/list/PathSandeshData[$i_nh]/nh/NhSandeshData/type)" t_nhs.xml`
103 local peer_type=`xmllint --xpath "string(//RouteUcSandeshData[src_ip='$dst_ip']/path_list/list/PathSandeshData[$i_nh]/peer)" t_nhs.xml`
104 if [ "$nh_type" = "interface" ]
105 then
106 echo "$comp_ip,$i_nh: I $peer_type"
107 #return
108 elif [ "$nh_type" = "tunnel" ]
109 then
110 echo "$comp_ip,$i_nh: T $peer_type"
111 #return
112 elif [ "$nh_type" = "vrf" ]
113 then
114 echo "$comp_ip,$i_nh: V $peer_type"
115 elif [[ "$nh_type" =~ .*"ECMP Composite".* ]] #[ "$nh_type" = "Composite" ]
116 then
117 local n_paths=`xmllint --xpath "count(//RouteUcSandeshData[src_ip='$dst_ip']/path_list/list/PathSandeshData[$i_nh]/nh/NhSandeshData/mc_list/list/McastData)" t_nhs.xml`
118 local n_null_paths=`xmllint --xpath "count(//RouteUcSandeshData[src_ip='$dst_ip']/path_list/list/PathSandeshData[$i_nh]/nh/NhSandeshData/mc_list/list/McastData[type='NULL'])" t_nhs.xml`
119 local n_paths=`expr $n_paths - $n_null_paths`
120 echo "$comp_ip,$i_nh: C$n_paths""N$n_null_paths $peer_type"
121 #return
122 fi
123 i_nh=`expr $i_nh + 1`
124 done
125 #echo "Eo"
126}
